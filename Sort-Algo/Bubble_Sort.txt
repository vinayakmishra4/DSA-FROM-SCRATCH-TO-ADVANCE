Procedure BubbleSort(arr, n)
    For i from 0 to n - 2 do
        For j from 0 to n - 2 - i do
            If arr[j] > arr[j + 1] then
                Swap arr[j] and arr[j + 1]
            End If
        End For
    End For
End Procedure
